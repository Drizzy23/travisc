document.addEventListener("DOMContentLoaded",function(){document.querySelector('.copy-btn').addEventListener('click',function(event){event.preventDefault();const addressText=document.querySelector('.coin-address').textContent;const tempInput=document.createElement('input');tempInput.style.position='absolute';tempInput.style.left='-9999px';tempInput.value=addressText;document.body.appendChild(tempInput);tempInput.select();document.execCommand('copy');document.body.removeChild(tempInput);alert('Address copied to clipboard!')})})
